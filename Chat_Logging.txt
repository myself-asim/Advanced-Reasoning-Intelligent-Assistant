```python
from datetime import datetime

def calculate_age(birth_date):
    """
    Calculate the age based on the birth date.
    
    Args:
        birth_date (str): The user's birth date in YYYY-MM-DD format.
    
    Returns:
        int: The user's age.
    """
    today = datetime.today()
    birth_date = datetime.strptime(birth_date, "%Y-%m-%d")
    age = today.year - birth_date.year - ((today.month, today.day) < (birth_date.month, birth_date.day))
    return age

def main():
    birth_date = input("Enter your birth date (YYYY-MM-DD): ")
    age = calculate_age(birth_date)
    print(f"You are {age} years old.")

if __name__ == "__main__":
    main()
```

1. Run the program.
2. Enter your birth date in YYYY-MM-DD format when prompted.
3. The program will display your exact age.

**Note:** This program assumes that the user's birth date is in the past. If the user's birth date is in the future, the program will not work correctly.
